main:	add	$a0,$s0,$zero
	add	$a1,$s1,$zero
	jal	sum2
	add	$s2,$v0,$zero

sum2:	addi $sp,$sp,-8
	sw $ra,4($sp)
	sw $a1, 0($sp)
	add $a1,$a0,$zero
	jal sum
	lw $a1, 0($sp)
	lw $ra, 4($sp)
	addi $sp, $sp, 8
	add $v0, $v0, $a1
	jr $ra

sum: 	addi $sp, $sp, -4
	sw $s0, 0($sp)
	add $s0, $a0, $a1
	add $v0,$s0,$zero
	lw $s0, 0($sp)
	addi $sp, $sp, 4
	jr $ra
